    final V putVal(int hash, K key, V value, boolean onlyIfAbsent,
                   boolean evict) {
    //node数组
    Node<K,V>[] tab; 
    //node对象
    Node<K,V> p;
    int n;
    int i;
    //判断node数组是否为空
    if ((tab = table) == null || (n = tab.length) == 0)
        //如果为空，调用resize方法
        n = (tab = resize()).length;
    // 如果【（数组长度-1）&hash】后的节点为空，则在该位置创建一个新节点
    if ((p = tab[i = (n - 1) & hash]) == null)
        tab[i] = newNode(hash, key, value, null);
    else {
         Node<K,V> e; 
         K k;
        //如果新节点的hash等于传入hash 并且 （当前节点的key值等于传入的key值， 或者 传入的key值不为空且传入的key值等于当前节点的key）
        if (p.hash == hash && ((k = p.key) == key || (key != null && key.equals(k))))
            //p节点复制给e
            e = p;
        //如果p是TreeNode节点，则调用TreeNode的putTreeVal方法
        else if (p instanceof TreeNode)
            e = ((TreeNode<K,V>)p).putTreeVal(this, tab, hash, key, value);
        //如果e=p.next为空，则将p.next创建一个新节点
        else {
            for (int binCount = 0; ; ++binCount) {
                if ((e = p.next) == null) {
                    p.next = newNode(hash, key, value, null);
                    如果
                    if (binCount >= TREEIFY_THRESHOLD - 1) // -1 for 1st
                        treeifyBin(tab, hash);
                    break;
                }
                if (e.hash == hash &&
                        ((k = e.key) == key || (key != null && key.equals(k))))
                    break;
                p = e;
            }
        }
        if (e != null) { // existing mapping for key
            V oldValue = e.value;
            if (!onlyIfAbsent || oldValue == null)
                e.value = value;
            afterNodeAccess(e);
            return oldValue;
        }
    }
    ++modCount;
    if (++size > threshold)
        resize();
    afterNodeInsertion(evict);
    return null;
}

final Node<K,V>[] resize() {
        Node<K,V>[] oldTab = table;
        //默认长度
        int oldCap = (oldTab == null) ? 0 : oldTab.length;
        //默认容积
        int oldThr = threshold;
        //新容积0
        int newCap, newThr = 0;
        if (oldCap > 0) {
            //如果旧容积大于等于默认最大容积，则设置容积为Integer的最大值，2３１　－　1
            if (oldCap >= MAXIMUM_CAPACITY) {
                threshold = Integer.MAX_VALUE;
                //返回node数组
                return oldTab;
            }
            //如果新数组长度等于旧数组的2倍，小于最大容积，而且旧有数组长度大于等于默认初始容量，则设置新容积为旧容积的2倍
            else if ((newCap = oldCap << 1) < MAXIMUM_CAPACITY &&
                     oldCap >= DEFAULT_INITIAL_CAPACITY)
                newThr = oldThr << 1; // double threshold
        }
        //如果旧有容积大于0
        else if (oldThr > 0) // initial capacity was placed in threshold
            //新长度等于旧有（默认）容积
            newCap = oldThr;
        //其他情况，新长度等于默认加载容积，新容积等于默认加载容积
        else {               // zero initial threshold signifies using defaults
            newCap = DEFAULT_INITIAL_CAPACITY;
            //新容积 = 默认负载因子*默认初始容积
            newThr = (int)(DEFAULT_LOAD_FACTOR * DEFAULT_INITIAL_CAPACITY);
        }
        //如果新容积等于0
        if (newThr == 0) {
            //新长度*负载因子
            float ft = (float)newCap * loadFactor;
            //新容积 = （如果新长度小于最大容积而且新长度*负载因子小于最大容积，则ft为新容积，若非最大容积，则）
            newThr = (newCap < MAXIMUM_CAPACITY && ft < (float)MAXIMUM_CAPACITY ?
                      (int)ft : Integer.MAX_VALUE);
        }
        //默认容积等于新容积
        threshold = newThr;
        //忽略警告
        @SuppressWarnings({"rawtypes","unchecked"})
        //使用新长度创建node数组        
        Node<K,V>[] newTab = (Node<K,V>[])new Node[newCap];
        //默认数组等于新数组
        table = newTab;
        //如果旧数组不为空，则遍历旧数组
        if (oldTab != null) {
            for (int j = 0; j < oldCap; ++j) {
                Node<K,V> e;
                //如果当前节点不为空，则将当前节点置为空，如果下一节点为空，新数组的节点（当前节点hash&（新长度-1））等于当前节点的值，
                if ((e = oldTab[j]) != null) {
                    oldTab[j] = null;
                    if (e.next == null)
                        newTab[e.hash & (newCap - 1)] = e;
                    //如果当前节点是treeNode类型，则调用split方法
                    else if (e instanceof TreeNode)
                        ((TreeNode<K,V>)e).split(this, newTab, j, oldCap);
                    //否则，
                    else { // preserve order
                        Node<K,V> loHead = null, 
                                loTail = null;
                        Node<K,V> hiHead = null, 
                                hiTail = null;
                        Node<K,V> next;
                        do {
                            next = e.next;
                            if ((e.hash & oldCap) == 0) {
                                if (loTail == null)
                                    loHead = e;
                                else
                                    loTail.next = e;
                                loTail = e;
                            }
                            else {
                                if (hiTail == null)
                                    hiHead = e;
                                else
                                    hiTail.next = e;
                                hiTail = e;
                            }
                        } while ((e = next) != null);
                        if (loTail != null) {
                            loTail.next = null;
                            newTab[j] = loHead;
                        }
                        if (hiTail != null) {
                            hiTail.next = null;
                            newTab[j + oldCap] = hiHead;
                        }
                    }
                }
            }
        }
        return newTab;
    }
